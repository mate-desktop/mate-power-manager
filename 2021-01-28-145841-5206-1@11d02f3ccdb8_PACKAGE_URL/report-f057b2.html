660/**
661 * gpm_manager_perform_policy:
662 * @manager: This class instance
663 * @policy: The policy that we should do, e.g. "suspend"
664 * @reason: The reason we are performing the policy action, e.g. "battery critical"
665 *
666 * Does one of the policy actions specified in the settings.
667 **/
668static gboolean
669gpm_manager_perform_policy (GpmManager *manager, const gchar *policy_key, const gchar *reason)
670{
671 GpmActionPolicy policy;
672
673 /* are we inhibited? */
674 if (gpm_manager_is_inhibit_valid (manager, FALSE(0), "policy action") == FALSE(0))
675 return FALSE(0);
676
677 policy = g_settings_get_enum (manager->priv->settings, policy_key);
678 g_debug ("action: %s set to %i (%s)", policy_key, policy, reason);
679
680 if (policy == GPM_ACTION_POLICY_NOTHING) {
681 g_debug ("doing nothing, reason: %s", reason);
682 } else if (policy == GPM_ACTION_POLICY_SUSPEND) {
683 gpm_manager_action_suspend (manager, reason);
684
685 } else if (policy == GPM_ACTION_POLICY_HIBERNATE) {
686 gpm_manager_action_hibernate (manager, reason);
687
688 } else if (policy == GPM_ACTION_POLICY_BLANK) {
689 gpm_manager_blank_screen (manager, NULL((void*)0));
690
691 } else if (policy == GPM_ACTION_POLICY_SHUTDOWN) {
692 g_debug ("shutting down, reason: %s", reason);
693 gpm_control_shutdown (manager->priv->control, NULL((void*)0));
694
695 } else if (policy == GPM_ACTION_POLICY_INTERACTIVE) {
696 GpmSession *session;
697 g_debug ("logout, reason: %s", reason);
698 session = gpm_session_new ();
699 gpm_session_logout (session);
700 g_object_unref (session);
701 } else {
702 g_warning ("unknown action %i", policy);
703 }
704
705 return TRUE(!(0));
706}

708/**
709 * gpm_manager_idle_do_sleep:
710 * @manager: This class instance
711 *
712 * This callback is called when we want to sleep. Use the users
713 * preference from the settings, but change it if we can't do the action.
714 **/
715static void
716gpm_manager_idle_do_sleep (GpmManager *manager)
717{
718 gboolean ret;
719 GError *error = NULL((void*)0);
720 GpmActionPolicy policy;
721
722 if (!manager->priv->on_battery)
723 policy = g_settings_get_enum (manager->priv->settings, GPM_SETTINGS_ACTION_SLEEP_TYPE_AC"action-sleep-type-ac");

725 policy = g_settings_get_enum (manager->priv->settings, GPM_SETTINGS_ACTION_SLEEP_TYPE_BATT"action-sleep-type-battery");
726
727 if (policy == GPM_ACTION_POLICY_NOTHING) {
728 g_debug ("doing nothing as system idle action");
729
730 } else if (policy == GPM_ACTION_POLICY_SUSPEND) {
731 g_debug ("suspending, reason: System idle");
732 ret = gpm_control_suspend (manager->priv->control, &error);
733 if (!ret) {
734 g_warning ("cannot suspend (error: %s), so trying hibernate", error->message);
735 g_error_free (error);

737 ret = gpm_control_hibernate (manager->priv->control, &error);
738 if (!ret) {
739 g_warning ("cannot suspend or hibernate: %s", error->message);
740 g_error_free (error);
741 }
742 }
743
744 } else if (policy == GPM_ACTION_POLICY_HIBERNATE) {
745 g_debug ("hibernating, reason: System idle");
746 ret = gpm_control_hibernate (manager->priv->control, &error);
747 if (!ret) {
748 g_warning ("cannot hibernate (error: %s), so trying suspend", error->message);
749 g_error_free (error);

751 ret = gpm_control_suspend (manager->priv->control, &error);
